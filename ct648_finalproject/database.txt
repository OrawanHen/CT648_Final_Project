CREATE TABLE quiz_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question VARCHAR(255) NOT NULL,
    correct_answer VARCHAR(255) NOT NULL,
    category VARCHAR(100),
    difficulty VARCHAR(50),
    type VARCHAR(50),
    incorrect_answers JSON NOT NULL
);
